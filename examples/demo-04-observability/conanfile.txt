# Demo-04 — Conan 2.x package list.
#
# Why Conan instead of system packages: rounds r28-r30 walked into
# UBI 9's repo gap one G-NN at a time (G-12 docker-compose
# delegation, G-13 EPEL needed, G-14 protobuf-devel still missing
# despite EPEL because UBI doesn't ship CRB). System packages on
# UBI for the modern C++ ecosystem are a moving target. Conan is
# the right answer architecturally — and §13 Reproducibility
# spends a chapter on exactly this lesson.
#
# This recipe pulls opentelemetry-cpp from Conan Center; that pulls
# gRPC, protobuf, abseil-cpp, c-ares and the rest as transitive
# deps. All static-linked into the binary so the runtime image
# doesn't need a single C++ ecosystem package.
#
# Why this exact version combo (G-24): rolling back from r42's
# opentelemetry-cpp/1.18.0 + r44's attempted grpc/1.62.0 pin.
# OTel-cpp 1.18.0's recipe **strictly requires** grpc/1.67.1 — the
# attempted grpc/1.62.0 pin produced "Conflict between grpc/1.67.1
# and grpc/1.62.0 in the graph. Conflict originates from
# opentelemetry-cpp/1.18.0." But auto-resolved grpc/1.67.1 has
# `grpc::Status::OK` and `grpc::GetGlobalCallbackHook()` removed
# from libgrpc++.a as linkable symbols (gRPC's 1.65+ ABI changes).
# Yet OTel-cpp's pre-built libopentelemetry_proto_grpc.a still
# references those symbols via gRPC's inline templates. Result:
# unresolvable undefined references, regardless of link order.
#
# This combination — opentelemetry-cpp/1.14.2 + grpc/1.62.0 +
# protobuf/3.21.12 + abseil/20240116.2 — is documented as
# working together in published guides (see e.g. OneUptime's
# Feb 2026 OpenTelemetry C++ dependency-management guide). The
# four packages were tested as a set. gRPC 1.62.0 has Status::OK
# and GetGlobalCallbackHook() defined as linkable statics; OTel-cpp
# 1.14.2's recipe accepts grpc/1.62.0 in its version range.
#
# Cost of the rollback: main.cpp may need adjustment because OTel-cpp
# 1.16.0 changed factory return types (api::TracerProvider →
# sdk::TracerProvider). Tracked separately if it surfaces.

[requires]
opentelemetry-cpp/1.14.2
grpc/1.62.0
protobuf/3.21.12
abseil/20240116.2

[generators]
CMakeDeps
CMakeToolchain

[generators]
CMakeDeps
CMakeToolchain

# No [layout] section: we use Conan 2.x's default flat layout because
# this is a single-build-type containerized one-shot compile. The
# `cmake_layout` pattern adds `build/<build_type>/generators/` path
# components that are useful for host-side multi-config workflows but
# just complicate the Containerfile path math here. With no [layout],
# `conan install . --output-folder=build/conan` puts the toolchain at
# `build/conan/conan_toolchain.cmake` — which is exactly what the
# Containerfile's `-DCMAKE_TOOLCHAIN_FILE=...` flag expects. See G-18.

[options]
# OTel-cpp options:
#  - with_otlp_grpc=True: the demo's exporter (OTLP/gRPC to lgtm:4317)
#  - with_otlp_http=False: the demo doesn't speak OTLP/HTTP
#  - with_zipkin=False: drops libcurl from the dep tree. libcurl is
#    pulled by OTel-cpp solely for the Zipkin exporter, which the
#    demo doesn't use. Skipping it eliminates an autotools-using
#    transitive dep that has its own perl-module + autotools needs
#    (G-17). About 5 fewer deps to build from source.
#  - shared=False: keep static so the runtime image is minimal.
opentelemetry-cpp/*:with_otlp_grpc=True
opentelemetry-cpp/*:with_otlp_http=False
opentelemetry-cpp/*:with_zipkin=False
opentelemetry-cpp/*:shared=False

# OpenSSL options: skip the FIPS module. gRPC pulls openssl as a
# transitive dep for TLS support, but our demo uses plaintext gRPC
# inside the container network — no TLS, no FIPS-validated crypto
# requirement. Skipping FIPS:
#   - removes the mk-fipsmodule-cnf.pl build step that needs perl
#     Digest::SHA (G-16)
#   - shrinks the openssl static library
#   - cuts openssl build time
openssl/*:no_fips=True

# Transitive deps: keep static linkage uniform. Wildcards apply to
# every package in the dep tree; recipes that don't have a `shared`
# option just ignore it.
*:shared=False
